query <- "
-- Seleciona os professores sem repeti-los (id e nome)
SELECT DISTINCT i.name

-- Seleciona a tabela de instrutores
FROM instructors i

-- Junta com a tabela de ensinamentos para saber quais seções cada professor lecionou
JOIN teachings t ON i.id = t.instructor_id

-- Junta com as seções para pegar os detalhes das turmas
JOIN sections s ON t.section_uuid = s.uuid

-- Junta com as ofertas de cursos para saber a qual curso cada seção pertence
JOIN course_offerings co ON s.course_offering_uuid = co.uuid

-- Junta com subject_memberships para saber a qual curso/disciplina pertence.
JOIN subject_memberships sm ON co.uuid = sm.course_offering_uuid

-- Junta com a tabela subjects para filtrar por estatística
JOIN subjects sub ON sm.subject_code = sub.code

-- Filtra apenas disciplinas cujo tópico é estatística
WHERE sub.abbreviation = 'STAT'
"

# Executar a query
professores_stat <- dbGetQuery(conn, query)

query_gpa <- 
  " -- Seleciona o id e o nome do professor, o nome da disciplina e calcula o GPA ponderado
SELECT 
  i.id AS instructor_id,             -- ID do professor (para identificar unicamente)
  i.name AS instructor_name,         -- Nome do professor
  c.name AS course_name,             -- Nome da disciplina/oferta de curso
  (
    -- Cálculo do GPA ponderado pela quantidade de alunos em cada nota
    (gd.a_count*4 + gd.ab_count*3.5 + gd.b_count*3 + gd.bc_count*2.5 
     + gd.c_count*2 + gd.d_count*1 + gd.f_count*0)
    /
    (gd.a_count + gd.ab_count + gd.b_count + gd.bc_count + gd.c_count + gd.d_count + gd.f_count)
  ) AS gpa                             -- GPA médio ponderado dessa oferta

-- Tabela principal de onde vêm as notas dos alunos
FROM grade_distributions gd

-- Conecta as notas às seções correspondentes
JOIN sections s 
  ON gd.section_number = s.number 
  AND gd.course_offering_uuid = s.course_offering_uuid

-- Conecta cada seção à oferta de curso correspondente
JOIN course_offerings co 
  ON s.course_offering_uuid = co.uuid

-- Liga cada oferta de curso ao assunto/disciplinas associadas
JOIN subject_memberships sm 
  ON co.uuid = sm.course_offering_uuid

-- Liga o código do assunto à tabela subjects para filtrar apenas STAT
JOIN subjects sub 
  ON sm.subject_code = sub.code

-- Liga cada seção aos professores que ministraram
JOIN teachings t 
  ON s.uuid = t.section_uuid

-- Liga o professor ao seu nome
JOIN instructors i 
  ON t.instructor_id = i.id

-- Liga a oferta de curso ao nome da disciplina
JOIN courses c 
  ON co.course_uuid = c.uuid

-- Filtra apenas disciplinas cujo tópico é Estatística
WHERE sub.abbreviation = 'STAT' "



gpa_stat <- dbGetQuery(conn, query_gpa)

# calcula professores e disciplinas mais difíceis/fáceis
resumo <- gpa_stat %>%
  # Professor mais difícil/fácil
  group_by(instructor_id, instructor_name) %>%
  summarise(mean_gpa = mean(gpa), .groups="drop") %>%
  summarise(
    prof_dificil = instructor_name[which.min(mean_gpa)],
    prof_facil = instructor_name[which.max(mean_gpa)]
  ) %>%
  # Transformar em tibble com todos os resultados
  bind_cols(
    gpa_stat %>%
      group_by(course_name) %>%
      summarise(mean_gpa = mean(gpa), .groups="drop") %>%
      summarise(
        disc_dificil = course_name[which.min(mean_gpa)],
        disc_facil = course_name[which.max(mean_gpa)]
      )
  ) %>%
  # Organizar em formato mais bonito visualmente
  tidyr::pivot_longer(everything(), names_to = "tipo", values_to = "nome")
